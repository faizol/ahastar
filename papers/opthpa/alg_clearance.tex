\begin{algorithm}
  \caption{A fast method for computing tile clearance}
  \label{alg-clearance}
  \begin{algorithmic}[1]
	\REQUIRE $map$
	\FOR{$x =$ width($map$); $x >= 1 $;$ x = x - 1$}
		\FOR{$y =$ height($map$)$; y >= 1 $;$ y = y -1$}	
			\STATE	cv($n$) = 0;
			\IF{$n \not \in obstacles$}
				\STATE $n \leftarrow$ tileAt(x, y);
				\STATE $i \leftarrow$ tileAt(x+1, y);
				\STATE $j \leftarrow$ tileAt(x, y+1);
				\STATE $k \leftarrow$ tileAt(x+1, y+1);
				\IF{$\lbrace i, j, k \rbrace \not \in obstacles$}
					\STATE cv($n$) = min(cv($i$), cv($j$), cv($k$)) + 1;
				\ELSE
					\STATE cv($n$) = 1;
				\ENDIF
			\ENDIF
		\ENDFOR
	\ENDFOR
  \end{algorithmic}
\end{algorithm}
